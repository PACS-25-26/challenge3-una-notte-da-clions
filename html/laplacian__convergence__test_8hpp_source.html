137 double Convergence_Test<solver_type, boundary_condition, funcType>::compute_error(const eigenMatrix& u_h, const eigenMatrix& X, const eigenMatrix& Y, funcType u_exact_lambda) {
138 double error_sum = 0.0;
139 unsigned n = u_h.rows();
140 double h = (data_original.x2 - data_original.x1) / (n - 1);
141 for (unsigned i = 0; i < n; ++i) {
142 for (unsigned j = 0; j < n; ++j) {
143 double u_exact = u_exact_lambda(X(i, j), Y(i, j));
144 error_sum += h*std::pow(u_h(i, j) - u_exact, 2);
145 }
146 }
147 return std::sqrt(error_sum);
