106
107 // Get rank information
108
109 // Distribute rows among processes. If the number of rows is not divided by the number of threads, the remaining ones will be assigned to the
110 // fisrt threads, as shown in Figure 1 of the challenge pdf file.
111 const unsigned local_rows = data.n / mpi_size;
112 const unsigned remainder_rows = data.n % mpi_size;
113
114 const unsigned start_row = mpi_rank * local_rows + std::min(remainder_rows, static_cast<unsigned>(mpi_rank));
115 const unsigned end_row = start_row + local_rows + (static_cast<unsigned>(mpi_rank) < remainder_rows ? 1 : 0);
116
117 meshX = eigenMatrix::Zero(end_row - start_row, data.n);
118 meshY = eigenMatrix::Zero(end_row - start_row, data.n);
119
120 #pragma omp parallel for
121 for(unsigned i = 0; i < end_row - start_row; i++) for(unsigned j = 0; j < data.n; j++){
122 meshX(i, j) = data.x1 + j * h;
123 meshY(i, j) = data.x1 + (i + start_row) * h;
124 }
